AppKit Headless allows developers to build a fully customizable ‚ÄúConnect‚Äù experience powered by AppKit‚Äôs existing features without using AppKit modal. Headless exposes wallet discovery, rendering WalletConnect QR, and social providers flow via simple hooks and utility functions.

## Installation

Headless requires AppKit to be configured with `enableHeadless` flag. See the [installation](../core/installation) docs before getting started and make sure you're enabling headless as expected.

```tsx
createAppKit({
  ...
  enableHeadless: true
})
```

## useAppKitWallets

The essential hook for accessing all type wallets and methods to build your custom connection user interface.

<CodeGroup>

```tsx useAppKitWallets.tsx
const { data, isFetchingWallets, isFetchingWcUri, wcUri, page, count, fetchWallets, connect } =
  useAppKitWallets()
```

</CodeGroup>

### Returns

<ResponseField name="data" type="WalletItem[]">
  List of all wallets (injected wallets and WalletConnect wallets combined).
</ResponseField>

<ResponseField name="isFetchingWallets" type="boolean">
  Boolean indicating if WalletConnect wallets are being fetched.
</ResponseField>

<ResponseField name="isFetchingWcUri" type="boolean">
  Boolean indicating if a WalletConnect URI is being fetched.
</ResponseField>

<ResponseField name="wcUri" type="string | undefined">
  The current WalletConnect URI for QR code display (set when connecting to a WalletConnect wallet).
</ResponseField>

<ResponseField name="page" type="number">
  The current page number of WalletConnect wallet results.
</ResponseField>

<ResponseField name="count" type="number">
  The total number of available wallets (for pagination/counting).
</ResponseField>

<ResponseField name="fetchWallets" type="(options) => Promise">
  Function to fetch WalletConnect wallets from the explorer API. Enables pagination/initial load.
</ResponseField>

<ResponseField name="connect" type="(wallet: WalletItem, namespace?: ChainNamespace) => Promise">
  Function to connect to a wallet. For WalletConnect wallets: starts connection & updates wcUri. For
  injected wallets: triggers direct connection.
</ResponseField>

# Examples

## Build a Connect UI

<Frame>
  <img src="/images/headless-connect.png" alt="Descriptive alt text" />
</Frame>

The `useAppKitWallets` hook returns list of injected and WalletConnect wallets let you render directly in the connect page.

If you configure your AppKit as multi-chain, we suggest rendering a namespace selection component where users should select which namespace to connect when using injected wallets. This could be achieved with the `connectors` array of the `WalletItem`.

<CodeGroup>

```tsx ConnectView.tsx
import React from 'react'
import { useAppKitWallets } from '@reown/appkit/react'

function ConnectView() {
  const { data } = useAppKitWallets()

  async function handleConnect(wallet: WalletItem, namespace?: ChainNamespace) {
    await connect(wallet, namespace)
      .then(() => {
        toast({ title: 'Connected', status: 'success' })
      })
      .catch(() => {
        toast({ title: 'Connection declined', status: 'error' })
        setConnectingWallet(undefined)
      })
  }

  return (
    <>
      {data.map(wallet => (
        <WalletListItem
          key={wallet.id}
          wallet={wallet}
          onConnect={() => {
            if (isMultiChain) {
              onOpenNamespaceDialog(item)
            } else {
              onConnect(wallet, wallet.connectors[0].chain)
            }
          }}
        />
      ))}
    </>
  )
}
```

```tsx WalletListItem.tsx
export function WalletListItem({ wallet, onConnect }) {
  return (
    <button onClick={onConnect}>
      <img url={wallet.imageUrl} />
      <span>{wallet.name}</span>
      {wallet.connectors.map(item => {
        return <img {item.chainImageUrl} alt={item.chain} />
      })}
    </button>
  )
}
```

```tsx NamespaceDialog.tsx
import { ChevronRightIcon } from 'lucide-react'
import Image from 'next/image'

import type { WalletItem } from '@reown/appkit'
import { ChainNamespace } from '@reown/appkit/networks'

import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle
} from '@/components/ui/dialog'
import { Item, ItemActions, ItemContent, ItemMedia, ItemTitle } from '@/components/ui/item'

interface NamespaceSelectionDialogProps {
  wallet: WalletItem | null
  onSelect?: (item: WalletItem, namespace: ChainNamespace) => void
}

const CHAIN_NAME_MAP = {
  eip155: 'EVM Networks',
  solana: 'Solana',
  polkadot: 'Polkadot',
  bip122: 'Bitcoin',
  cosmos: 'Cosmos',
  sui: 'Sui',
  stacks: 'Stacks',
  ton: 'TON'
}

export function NamespaceSelectionDialog({ wallet, onSelect }: NamespaceSelectionDialogProps) {
  return (
    <Dialog>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Connect to {wallet.name}</DialogTitle>
          <DialogDescription>
            {wallet.name} supports multiple chains. Select a chain you want to connect to.
          </DialogDescription>
        </DialogHeader>
        <div className="grid gap-2">
          {wallet.connectors.map(connector => (
            <Item onClick={() => onSelect(wallet, connector.chain)}>
              {connector.chainImageUrl && (
                <ItemMedia className="rounded-md overflow-hidden w-8 h-8">
                  <Image
                    src={connector.chainImageUrl}
                    alt={connector.chain}
                    width={32}
                    height={32}
                  />
                </ItemMedia>
              )}
              <ItemContent>
                <ItemTitle>{CHAIN_NAME_MAP[connector.chain]}</ItemTitle>
              </ItemContent>
              <ItemActions>
                <ChevronRightIcon className="size-4" />
              </ItemActions>
            </Item>
          ))}
        </div>
      </DialogContent>
    </Dialog>
  )
}
```

## Search and Connect to WalletConnect Wallets

<Frame>
  <img src="/images/headless-wc.png" alt="Descriptive alt text" />
</Frame>

WalletConnect supports 500+ wallets. You can let users to search their own wallets to connect to your app.

<CodeGroup>

```tsx AllWalletsView.tsx
import React from 'react'
import { useAppKitWallets } from '@reown/appkit/react'

function AllWalletsView() {
  const { data, isFetchingWcUri, isFetchingWallets, fetchWallets, connect } = useAppKitWallets()
  const [inputValue, setInputValue] = useState('')
  const searchQuery = useDebounceValue(inputValue, 500)

  const wcWallets = data.filter(w => !w.isInjected)

  useEffect(() => {
    fetchWallets?.()
  }, [])

  useEffect(() => {
    if (searchQuery.length > 0) {
      fetchWallets?.({ query: searchQuery })
    } else {
      fetchWallets?.()
    }
  }, [searchQuery])

  return wcWallets.map(wallet => {
    return <WalletListItem wallet={wallet} onConnect={connect} />
  })
}
```

```tsx WalletListItem.tsx
export function WalletListItem({ wallet, onConnect }) {
  return (
    <button onClick={onConnect}>
      <img url={wallet.imageUrl} />
      <span>{wallet.name}</span>
      {wallet.connectors.map(item => {
        return <img {item.chainImageUrl} alt={item.chain} />
      })}
    </button>
  )
}
```

</CodeGroup>

If user select a WalletConnect wallet, you can show the QR code to the users and let them scan it with their preferred wallets. To handle the QR flows, you can use the `isFetchingWcUri` and `wcUri` props from the `useAppKitWallets` hook.

<CodeGroup>

```tsx QrCodeView.tsx
import QRCode from 'react-qr-code'

import { Loader2Icon } from 'lucide-react'

import type { WalletItem } from '@reown/appkit'
import { useAppKitWallets } from '@reown/appkit/react'

interface QrCodeViewProps {
  wallet: WalletItem
}

export function QrCodeView({ wallet }: QrCodeViewProps) {
  const { wcUri, isFetchingWcUri } = useAppKitWallets()

  return (
    <div>
      <img src={wallet.imageUrl} alt={wallet.name} width={40} height={40} />
      <span className="font-medium">{wallet.name}</span>

      <div>
        {isFetchingWcUri ? (
          <LoaderIcon className="animate-spin" />
        ) : wcUri ? (
          <QRCode
            value={wcUri}
            size={200}
            style={{ height: 'auto', maxWidth: '100%', width: 'auto' }}
          />
        ) : null}
      </div>
    </div>
  )
}
```
